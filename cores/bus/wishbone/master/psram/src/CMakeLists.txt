
add_custom_command (
  OUTPUT  VPsramWishboneMaster.cpp
          VPsramWishboneMaster.h
          VPsramWishboneMaster__Syms.cpp
          VPsramWishboneMaster__Syms.h
          VPsramWishboneMaster__Trace.cpp
          VPsramWishboneMaster__Trace__Slow.cpp
  DEPENDS PsramWishboneMaster.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/PsramWishboneMaster.v
)


add_library (VPsramWishboneMaster STATIC
    VPsramWishboneMaster.cpp
    VPsramWishboneMaster__Syms.cpp
    VPsramWishboneMaster__Trace.cpp
    VPsramWishboneMaster__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    PsramWishboneMaster.v
)

target_compile_options(VPsramWishboneMaster PRIVATE ${VK_CPPFLAGS})

target_include_directories(VPsramWishboneMaster PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VPsramWishboneMaster verilated)
