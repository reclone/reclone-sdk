
add_custom_command (
  OUTPUT  VVideoFormatTiming.cpp
          VVideoFormatTiming.h
          VVideoFormatTiming__Syms.cpp
          VVideoFormatTiming__Syms.h
          VVideoFormatTiming__Trace.cpp
          VVideoFormatTiming__Trace__Slow.cpp
  DEPENDS VideoFormatTiming.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoFormatTiming.v
)


add_library (VVideoFormatTiming STATIC
    VVideoFormatTiming.cpp
    VVideoFormatTiming__Syms.cpp
    VVideoFormatTiming__Trace.cpp
    VVideoFormatTiming__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoFormatTiming.v
)

target_compile_options(VVideoFormatTiming PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoFormatTiming PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoFormatTiming verilated)
