
add_custom_command (
  OUTPUT  VTextScreenRam.cpp
          VTextScreenRam.h
          VTextScreenRam__Syms.cpp
          VTextScreenRam__Syms.h
          VTextScreenRam__Trace.cpp
          VTextScreenRam__Trace__Slow.cpp
  DEPENDS TextScreenRam.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/TextScreenRam.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)


add_library (VTextScreenRam STATIC
    VTextScreenRam.cpp
    VTextScreenRam__Syms.cpp
    VTextScreenRam__Trace.cpp
    VTextScreenRam__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    TextScreenRam.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)

target_compile_options(VTextScreenRam PRIVATE ${VK_CPPFLAGS})

target_include_directories(VTextScreenRam PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VTextScreenRam verilated)
