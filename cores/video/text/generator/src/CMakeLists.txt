
add_custom_command (
  OUTPUT  VTextOverlayGenerator.cpp
          VTextOverlayGenerator.h
          VTextOverlayGenerator__Syms.cpp
          VTextOverlayGenerator__Syms.h
          VTextOverlayGenerator__Trace.cpp
          VTextOverlayGenerator__Trace__Slow.cpp
  DEPENDS TextOverlayGenerator.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/TextOverlayGenerator.v
)


add_library (VTextOverlayGenerator STATIC
    VTextOverlayGenerator.cpp
    VTextOverlayGenerator__Syms.cpp
    VTextOverlayGenerator__Trace.cpp
    VTextOverlayGenerator__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    TextOverlayGenerator.v
)

target_compile_options(VTextOverlayGenerator PRIVATE ${VK_CPPFLAGS})

target_include_directories(VTextOverlayGenerator PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VTextOverlayGenerator verilated)
