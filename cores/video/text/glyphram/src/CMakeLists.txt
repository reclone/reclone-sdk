
add_custom_command (
  OUTPUT  VTextGlyphRam.cpp
          VTextGlyphRam.h
          VTextGlyphRam__Syms.cpp
          VTextGlyphRam__Syms.h
          VTextGlyphRam__Trace.cpp
          VTextGlyphRam__Trace__Slow.cpp
  DEPENDS TextGlyphRam.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/TextGlyphRam.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)


add_library (VTextGlyphRam STATIC
    VTextGlyphRam.cpp
    VTextGlyphRam__Syms.cpp
    VTextGlyphRam__Trace.cpp
    VTextGlyphRam__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    TextGlyphRam.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)

target_compile_options(VTextGlyphRam PRIVATE ${VK_CPPFLAGS})

target_include_directories(VTextGlyphRam PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VTextGlyphRam verilated)
