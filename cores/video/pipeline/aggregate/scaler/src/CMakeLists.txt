
add_custom_command (
  OUTPUT  VAggregateVideoScaler.cpp
          VAggregateVideoScaler.h
          VAggregateVideoScaler__Syms.cpp
          VAggregateVideoScaler__Syms.h
          VAggregateVideoScaler__Trace.cpp
          VAggregateVideoScaler__Trace__Slow.cpp
  DEPENDS AggregateVideoScaler.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/AggregateVideoScaler.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
    ${PROJECT_SOURCE_DIR}/video/pipeline/filter/integerscale/src/VideoIntegerScale.v
)


add_library (VAggregateVideoScaler STATIC
    VAggregateVideoScaler.cpp
    VAggregateVideoScaler__Syms.cpp
    VAggregateVideoScaler__Trace.cpp
    VAggregateVideoScaler__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    AggregateVideoScaler.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
    ${PROJECT_SOURCE_DIR}/video/pipeline/filter/integerscale/src/VideoIntegerScale.v
)

target_compile_options(VAggregateVideoScaler PRIVATE ${VK_CPPFLAGS})

target_include_directories(VAggregateVideoScaler PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VAggregateVideoScaler verilated)
