
add_custom_command (
  OUTPUT  VVideoTimingSink.cpp
          VVideoTimingSink.h
          VVideoTimingSink__Syms.cpp
          VVideoTimingSink__Syms.h
          VVideoTimingSink__Trace.cpp
          VVideoTimingSink__Trace__Slow.cpp
  DEPENDS VideoTimingSink.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoTimingSink.v
    ${PROJECT_SOURCE_DIR}/video/timing/src/VideoFormatTiming.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/async/src/AsyncFifo.v
    ${PROJECT_SOURCE_DIR}/counter/gray/src/GrayCounter.v
)


add_library (VVideoTimingSink STATIC
    VVideoTimingSink.cpp
    VVideoTimingSink__Syms.cpp
    VVideoTimingSink__Trace.cpp
    VVideoTimingSink__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoTimingSink.v
    ${PROJECT_SOURCE_DIR}/video/timing/src/VideoFormatTiming.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/async/src/AsyncFifo.v
    ${PROJECT_SOURCE_DIR}/counter/gray/src/GrayCounter.v
)

target_compile_options(VVideoTimingSink PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoTimingSink PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoTimingSink verilated)
