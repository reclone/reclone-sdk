
add_custom_command (
  OUTPUT  VVideoHorizontalStretch.cpp
          VVideoHorizontalStretch.h
          VVideoHorizontalStretch__Syms.cpp
          VVideoHorizontalStretch__Syms.h
          VVideoHorizontalStretch__Trace.cpp
          VVideoHorizontalStretch__Trace__Slow.cpp
  DEPENDS VideoHorizontalStretch.v ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoHorizontalStretch.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)


add_library (VVideoHorizontalStretch STATIC
    VVideoHorizontalStretch.cpp
    VVideoHorizontalStretch__Syms.cpp
    VVideoHorizontalStretch__Trace.cpp
    VVideoHorizontalStretch__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoHorizontalStretch.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)

target_compile_options(VVideoHorizontalStretch PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoHorizontalStretch PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoHorizontalStretch verilated)
