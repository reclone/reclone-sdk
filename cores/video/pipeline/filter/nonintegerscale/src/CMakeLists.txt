
add_custom_command (
  OUTPUT  VVideoNonIntegerScale.cpp
          VVideoNonIntegerScale.h
          VVideoNonIntegerScale__Syms.cpp
          VVideoNonIntegerScale__Syms.h
          VVideoNonIntegerScale__Trace.cpp
          VVideoNonIntegerScale__Trace__Slow.cpp
  DEPENDS VideoNonIntegerScale.v ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoNonIntegerScale.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)


add_library (VVideoNonIntegerScale STATIC
    VVideoNonIntegerScale.cpp
    VVideoNonIntegerScale__Syms.cpp
    VVideoNonIntegerScale__Trace.cpp
    VVideoNonIntegerScale__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoNonIntegerScale.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)

target_compile_options(VVideoNonIntegerScale PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoNonIntegerScale PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoNonIntegerScale verilated)
