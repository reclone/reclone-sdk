
add_custom_command (
  OUTPUT  VVideoBinningFilter.cpp
          VVideoBinningFilter.h
          VVideoBinningFilter__Syms.cpp
          VVideoBinningFilter__Syms.h
          VVideoBinningFilter__Trace.cpp
          VVideoBinningFilter__Trace__Slow.cpp
  DEPENDS VideoBinningFilter.v ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoBinningFilter.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)


add_library (VVideoBinningFilter STATIC
    VVideoBinningFilter.cpp
    VVideoBinningFilter__Syms.cpp
    VVideoBinningFilter__Trace.cpp
    VVideoBinningFilter__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoBinningFilter.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
    ${PROJECT_SOURCE_DIR}/memory/blockram/dual-port/src/BlockRamDualPort.v
)

target_compile_options(VVideoBinningFilter PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoBinningFilter PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoBinningFilter verilated)
