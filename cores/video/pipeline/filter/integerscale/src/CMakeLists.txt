
add_custom_command (
  OUTPUT  VVideoIntegerScale.cpp
          VVideoIntegerScale.h
          VVideoIntegerScale__Syms.cpp
          VVideoIntegerScale__Syms.h
          VVideoIntegerScale__Trace.cpp
          VVideoIntegerScale__Trace__Slow.cpp
  DEPENDS VideoIntegerScale.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoIntegerScale.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)


add_library (VVideoIntegerScale STATIC
    VVideoIntegerScale.cpp
    VVideoIntegerScale__Syms.cpp
    VVideoIntegerScale__Trace.cpp
    VVideoIntegerScale__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoIntegerScale.v
    ${PROJECT_SOURCE_DIR}/memory/fifo/sync/src/SyncFifo.v
)

target_compile_options(VVideoIntegerScale PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoIntegerScale PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoIntegerScale verilated)
