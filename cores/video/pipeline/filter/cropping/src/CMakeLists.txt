
add_custom_command (
  OUTPUT  VVideoCroppingFilter.cpp
          VVideoCroppingFilter.h
          VVideoCroppingFilter__Syms.cpp
          VVideoCroppingFilter__Syms.h
          VVideoCroppingFilter__Trace.cpp
          VVideoCroppingFilter__Trace__Slow.cpp
  DEPENDS VideoCroppingFilter.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoCroppingFilter.v
)


add_library (VVideoCroppingFilter STATIC
    VVideoCroppingFilter.cpp
    VVideoCroppingFilter__Syms.cpp
    VVideoCroppingFilter__Trace.cpp
    VVideoCroppingFilter__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoCroppingFilter.v
)

target_compile_options(VVideoCroppingFilter PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoCroppingFilter PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoCroppingFilter verilated)
