
add_custom_command (
  OUTPUT  VVideoGeneratorSource.cpp
          VVideoGeneratorSource.h
          VVideoGeneratorSource__Syms.cpp
          VVideoGeneratorSource__Syms.h
          VVideoGeneratorSource__Trace.cpp
          VVideoGeneratorSource__Trace__Slow.cpp
  DEPENDS VideoGeneratorSource.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/VideoGeneratorSource.v
)


add_library (VVideoGeneratorSource STATIC
    VVideoGeneratorSource.cpp
    VVideoGeneratorSource__Syms.cpp
    VVideoGeneratorSource__Trace.cpp
    VVideoGeneratorSource__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    VideoGeneratorSource.v
)

target_compile_options(VVideoGeneratorSource PRIVATE ${VK_CPPFLAGS})

target_include_directories(VVideoGeneratorSource PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VVideoGeneratorSource verilated)
