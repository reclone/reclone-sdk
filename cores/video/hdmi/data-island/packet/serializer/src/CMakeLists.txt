
add_custom_command (
  OUTPUT  VDataIslandPacketSerializer.cpp
          VDataIslandPacketSerializer.h
          VDataIslandPacketSerializer__Syms.cpp
          VDataIslandPacketSerializer__Syms.h
          VDataIslandPacketSerializer__Trace.cpp
          VDataIslandPacketSerializer__Trace__Slow.cpp
  DEPENDS DataIslandPacketSerializer.v
    ${PROJECT_SOURCE_DIR}/video/hdmi/data-island/ecc/src/BchEccSingleBitEncoder.v
    ${PROJECT_SOURCE_DIR}/video/hdmi/data-island/ecc/src/BchEccDualBitEncoder.v

  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/DataIslandPacketSerializer.v
    ${PROJECT_SOURCE_DIR}/video/hdmi/data-island/ecc/src/BchEccSingleBitEncoder.v
    ${PROJECT_SOURCE_DIR}/video/hdmi/data-island/ecc/src/BchEccDualBitEncoder.v
    ${PROJECT_SOURCE_DIR}/video/hdmi/data-island/ecc/src/BchEccStep.v
)

    VDataIslandPacketSerializer__Trace.cpp
    VDataIslandPacketSerializer__Trace__Slow.cpp
    DataIslandPacketSerializer.v
)
