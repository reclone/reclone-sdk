
add_custom_command (
  OUTPUT  VCpu65xxMicroArchitecture.cpp
          VCpu65xxMicroArchitecture.h
          VCpu65xxMicroArchitecture__Syms.cpp
          VCpu65xxMicroArchitecture__Syms.h
          VCpu65xxMicroArchitecture__Trace.cpp
          VCpu65xxMicroArchitecture__Trace__Slow.cpp
  DEPENDS Cpu65xxMicroArchitecture.v
          ${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src/Cpu65xxMicrocodeConstants.vh
          ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -I${CMAKE_CURRENT_SOURCE_DIR} -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/Cpu65xxMicroArchitecture.v
    ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
)


add_library (VCpu65xxMicroArchitecture STATIC
    VCpu65xxMicroArchitecture.cpp
    VCpu65xxMicroArchitecture__Syms.cpp
    VCpu65xxMicroArchitecture__Trace.cpp
    VCpu65xxMicroArchitecture__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    Cpu65xxMicroArchitecture.v
    ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
    ${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src/Cpu65xxMicrocodeConstants.vh
)

target_compile_options(VCpu65xxMicroArchitecture PRIVATE ${VK_CPPFLAGS})

target_include_directories(VCpu65xxMicroArchitecture PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VCpu65xxMicroArchitecture verilated)
