
add_custom_command (
  OUTPUT  VCpu65xxAlu.cpp
          VCpu65xxAlu.h
          VCpu65xxAlu__Syms.cpp
          VCpu65xxAlu__Syms.h
          VCpu65xxAlu__Trace.cpp
          VCpu65xxAlu__Trace__Slow.cpp
  DEPENDS Cpu65xxAlu.v ${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src/Cpu65xxMicrocodeConstants.vh
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} -I${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/Cpu65xxAlu.v
)


add_library (VCpu65xxAlu STATIC
    VCpu65xxAlu.cpp
    VCpu65xxAlu__Syms.cpp
    VCpu65xxAlu__Trace.cpp
    VCpu65xxAlu__Trace__Slow.cpp
    # Verilog .v files are added here so that they get added to any IDE project generated by CMake.
    Cpu65xxAlu.v
)

target_compile_options(VCpu65xxAlu PRIVATE ${VK_CPPFLAGS})

target_include_directories(VCpu65xxAlu PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VCpu65xxAlu verilated)
