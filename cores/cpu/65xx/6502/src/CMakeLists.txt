
add_custom_command (
  OUTPUT  VCpu6502.cpp
          VCpu6502.h
          VCpu6502__Syms.cpp
          VCpu6502__Syms.h
          VCpu6502__Trace.cpp
          VCpu6502__Trace__Slow.cpp
  DEPENDS Cpu6502.v
          Cpu6502MicrocodeRom.v
          Cpu6502Opcodes.vh
          ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
          ${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src/Cpu65xxMicrocodeConstants.vh
  COMMAND perl ${VERILATOR} ${VERILATOR_OPTS} 
    -I${CMAKE_CURRENT_SOURCE_DIR}
    -I${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src
    -Mdir ${CMAKE_CURRENT_BINARY_DIR} -cc
    ${CMAKE_CURRENT_SOURCE_DIR}/Cpu6502.v
    ${CMAKE_CURRENT_SOURCE_DIR}/Cpu6502MicrocodeRom.v
    ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
)

# Verilog .v files are added here so that they get added to any IDE project generated by CMake.
add_library (VCpu6502 STATIC
    VCpu6502.cpp
    VCpu6502__Syms.cpp
    VCpu6502__Trace.cpp
    VCpu6502__Trace__Slow.cpp
    Cpu6502.v
    Cpu6502MicrocodeRom.v
    ${PROJECT_SOURCE_DIR}/cpu/65xx/alu/src/Cpu65xxAlu.v
    Cpu6502Opcodes.vh
    ${PROJECT_SOURCE_DIR}/cpu/65xx/uarch/src/Cpu65xxMicrocodeConstants.vh
)

target_compile_options(VCpu6502 PRIVATE ${VK_CPPFLAGS})

target_include_directories(VCpu6502 PUBLIC ${CMAKE_CURRENT_BINARY_DIR})

target_link_libraries(VCpu6502 verilated)
